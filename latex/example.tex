  \oner{\vdash\one}

  \parrr{\vdash (\bot\plus\bot)\parr A\orth, \top\tensor A}
  \der{\vdash \wn{((\bot\plus\bot)\parr A\orth)}, \top\tensor A}
  \ocr{\vdash \wn{((\bot\plus\bot)\parr A\orth)}, \oc{(\top\tensor A)}}
  \wkr{\vdash \wn{((\bot\plus\bot)\parr A\orth)}, \wn{((\bot\plus\bot)\parr A\orth)}, \oc{(\top\tensor A)}}
  \cor{\vdash \wn{((\bot\plus\bot)\parr A\orth)}, \oc{(\top\tensor A)}}
\end{prooftree*}
